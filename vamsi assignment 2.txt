1.SELECT DEPTNO
  FROM EMP
  WHERE EMPNO IS NULL ;

2.SELECT ENAME,JOB
  FROM EMP
  WHERE COMM IS NULL ;

3.SELECT ENAME,JOB
  FROM EMP,DEPT
  WHERE COMM>1000 AND LOC LIKE 'NEWYORK';

4.SELECT COUNT(EMPNO)
  FROM EMP,DEPT
  WHERE LOC LIKE 'CHICAGO';

5.SELECT *
  FROM EMP,DEPT
  WHERE LOC LIKE 'CHICAGO';

6.SELECT ENAME,LOC
  FROM EMP,DEPT
  ORDER BY LOC;

7.SELECT COUNT(DISTINCT EMPNO) AS NOOFEMP,COUNT(DISTINCT DEPTNO) AS NOOFDEPT
  FROM EMP;

8.SELECT COUNT(MGR) AS NOOFMGR
   FROM EMP;

   FROM EMP,DEPT

13.SELECT EMPNO,ENAME,JOB,MGR,HIREDATE,SAL,COMM,DEPTNO,NULL AS LOC
   FROM EMP
   UNION
   SELECT NULL,NULL,DNAME AS JOB ,NULL,NULL,NULL,NULL,DEPTNO,LOC
   FROM DEPT;

   CASE
   WHEN JOB != 'SALES'

   END 
   FROM EMP;

   FROM EMP
   WHERE SAL<(SELECT SAL
              FROM EMP
